struct node{
       int data;
       struct node *prev, *next;
}*start = NULL;

static int node_count = 0;

/* Insert a new node at the end of the list */
void insertNode(int data) {
        struct node *temp, *ptr;
       ptr = (struct node *)malloc(sizeof (struct node));
	ptr->data = data;
        ptr->prev = NULL;
        ptr->next = NULL;

        if (start==NULL) {
                start = ptr;
                node_count++;
                return;
        } else {
                temp = start;
                while (temp) {
                        if (temp->next == NULL) {
                                temp->next = ptr;
                                ptr->prev = temp;
                                node_count++;
                                return;
                        }
                        temp=temp->next;
                }
        }
}
